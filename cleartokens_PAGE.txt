<h2>Clear Subscriber Tokens</h2>
<p style="color:#888;font-size:0.85rem">This deletes all FCM tokens so you stop getting duplicate notifications.<br>Re-enable notifications on your phone after clearing.</p>
<input type="password" id="pw" placeholder="Admin password">
<br>
<button onclick="go()">Clear All Tokens</button>
<div id="result"></div>
<script>
function go() {
  var pw = document.getElementById('pw').value.trim();
  if (!pw) { show('Enter your password', 'err'); return; }
  var fd = new FormData();
  fd.append('password', pw);
  fd.append('action', 'cleartokens');
  document.querySelector('button').textContent = 'Clearing...';
  fetch('/api/send', { method: 'POST', body: fd })
    .then(function(r) { return r.json().then(function(d) { return {s:r.status, d:d}; }); })
    .then(function(r) {
      document.querySelector('button').textContent = 'Clear All Tokens';
      if (r.d.ok) show('Deleted ' + r.d.deleted + ' token(s). Now re-enable notifications on your phone.', 'ok');
      else if (r.s === 401) show('Wrong password. This must match the password you use to send posts.', 'err');
      else show('Error: ' + JSON.stringify(r.d), 'err');
    })
    .catch(function(e) {
      document.querySelector('button').textContent = 'Clear All Tokens';
      show('Network error: ' + e.message, 'err');
    });
}
function show(msg, cls) { var el = document.getElementById('result'); el.textContent = msg; el.className = cls; }
</script>
